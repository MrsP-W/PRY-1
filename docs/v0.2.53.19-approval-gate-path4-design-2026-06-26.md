## 四、4 类动作实际写入流程(路径 4 设计)

### 4.1 `outbox.approve` 实际写入

```python
# handler._execute_real_write(action, target_id, audit)
def _approve_outbox_impl(target_id: str, *, audit: AuditContext) -> WriteResult:
    """路径 4 实际写入 — 调 OutboxStore.update_status(PENDING_SEND → APPROVED).

    异常收窄(沿 note_confirm_service.py:113-115 + v0.2.53.14 §7.3):
        - OutboxIllegalTransitionError 异常必须透传
        - 不收容(用户主动操作)
    """
    if not self._outbox_store:
        return WriteResult(success=False, affected_id=None, error="writer_not_initialized", ...)
    try:
        outbox_id = int(target_id)
    except (TypeError, ValueError):
        return WriteResult(success=False, affected_id=None, error="invalid_target_id", ...)
    last_approved_at_ms = _now_ms()
    entry = self._outbox_store.update_status(
        outbox_id,
        OutboxStatus.APPROVED,
        from_status=OutboxStatus.PENDING_SEND,
        last_approved_at_ms=last_approved_at_ms,
    )
    return WriteResult(
        success=True,
        affected_id=str(entry.id),
        error=None,
        reason=f"outbox {entry.id} approved at {last_approved_at_ms}",
        audit_id=None,
        write_executed=True,
    )
```

### 4.2 `outbox.cancel` 实际写入

```python
def _cancel_outbox_impl(target_id: str, *, audit: AuditContext) -> WriteResult:
    """路径 4 实际写入 — 调 OutboxStore.update_status(* → CANCELLED)."""
    if not self._outbox_store:
        return WriteResult(success=False, affected_id=None, error="writer_not_initialized", ...)
    outbox_id = int(target_id)
    entry = self._outbox_store.update_status(
        outbox_id,
        OutboxStatus.CANCELLED,
        from_status=OutboxStatus.PENDING_SEND,  # 也允许 APPROVED
        last_approved_at_ms=None,  # 取消不更新审批时间戳
    )
    return WriteResult(success=True, affected_id=str(entry.id), ...)
```

### 4.3 `notes.confirm` 实际写入

```python
def _confirm_note_impl(target_id: str, *, audit: AuditContext) -> WriteResult:
    """路径 4 实际写入 — 调 NoteConfirmServiceImpl.confirm_note.

    异常收窄(沿 note_confirm_service.py:113-115):用户主动操作异常必须透传.
    """
    if not self._note_confirm_service:
        return WriteResult(success=False, affected_id=None, error="writer_not_initialized", ...)
    self._note_confirm_service.confirm_note(target_id)
    return WriteResult(success=True, affected_id=target_id, ...)
```

### 4.4 `finance.dismiss_anomaly` 实际写入

```python
def _dismiss_anomaly_impl(target_id: str, *, audit: AuditContext) -> WriteResult:
    """路径 4 实际写入 — 调 AnomalyDismissalServiceImpl.dismiss(anomaly_id, reason=audit.reason)."""
    if not self._anomaly_dismissal_service:
        return WriteResult(success=False, affected_id=None, error="writer_not_initialized", ...)
    dismissal = self._anomaly_dismissal_service.dismiss(
        target_id,
        reason=audit.reason or "",
    )
    if not dismissal.success:
        return WriteResult(
            success=False,
            affected_id=None,
            error=dismissal.error or "dismiss_failed",
            reason=dismissal.reason,
        )
    return WriteResult(
        success=True,
        affected_id=dismissal.anomaly_id,
        error=None,
        reason=dismissal.reason or "",
        write_executed=True,
    )
```
