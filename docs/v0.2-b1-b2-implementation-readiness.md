def _compute_sla_due_at_ms(self, priority: str, created_at_ms: int) -> int:
    """预计算 SLA 截止时间(created_at + sla_threshold_ms(priority))。"""
    from my_ai_employee.scheduler.sla import _SLA_THRESHOLDS
    threshold_ms, _ = _SLA_THRESHOLDS[priority]
    return created_at_ms + threshold_ms
