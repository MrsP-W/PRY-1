**写入日志**:

```python
logger.info(
    "set_smtp_password_for_provider(%s, %s, %s)",
    provider,
    redact_email(email),                 # alice@outlook.com → al***@outlook.com

```

**读取日志**(避免显示 password):
```python
# 安全写法(只显示 provider + email)
result = keychain.get_smtp_password_for_provider("outlook", "alice@outlook.com")
if result.ok:
    logger.info("get_smtp_password_for_provider(outlook, %s) → ok", redact_email("alice@outlook.com"))
    return result.value
else:
    logger.warning("get_smtp_password_for_provider(outlook, %s) → fail: %s",
```

### 3.2 脱敏方法实现(`scripts/check_keychain_redaction.py`)

```python
def redact_email(email: str) -> str:
    """邮箱脱敏 — 保留前 2 字符 + @ + 域名后缀。"""
    if "@" not in email:
        return "***"
    local, domain = email.split("@", 1)
    if len(local) <= 2:
        redacted_local = "*" * len(local)
    else:
        redacted_local = local[:2] + "***"
    return f"{redacted_local}@{domain}"


def redact_token(token: str) -> str:
    """OAuth token 脱敏 — 保留前 4 + 后 4 字符 + 中间星号。"""
    if not token or len(token) <= 8:
        return "***"
    return f"{token[:4]}***{token[-4:]}"


def redact_password(password: str) -> str:
    """密码/授权码脱敏 — 全星号 + 长度。"""
    if not password:
        return "***"
    return f"{'*' * min(len(password), 8)}(len={len(password)})"
```
