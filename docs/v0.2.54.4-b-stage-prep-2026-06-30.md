### 4.3 spike 草稿生成 fixture(伪代码)

```python
# tests/dashboard/spike_fixtures/path4_spike_100.py
# 沿 v0.2.53.50 reports UX 范本

def generate_path4_spike_100() -> list[Path4SpikeItem]:
    """生成 100 个 Path 4 spike 草稿(分批 10+20+30+40).

    Returns:
        100 个 Path4SpikeItem,每个含:
        - action: approve_outbox / confirm_note / dismiss_anomaly
        - target_id: faker 生成的合法 target_id
        - actor: 测试 actor
        - reason: 测试 reason
        - expected_audit: 预期 audit_store._records 中应有 1 条
    """
    items = []
    for batch_size in [10, 20, 30, 40]:
        for _ in range(batch_size):
            action = faker.random_element(["approve_outbox", "confirm_note", "dismiss_anomaly"])
            items.append(Path4SpikeItem(
                action=action,
                target_id=faker.uuid4(),
                actor=faker.name(),
                reason=faker.sentence(),
                expected_audit=True,
            ))
    return items
```

### 4.4 spike 验证脚本(伪代码)

```python
# scripts/run_path4_spike.py
# 沿 D5.6.4 100 封 spike 脚本范本

def main():
    """100 封 Path 4 spike 验证.

    前置:
    - 5 门全开(DASHBOARD_WRITE_API=1 / confirm_text=CONFIRM_WRITE / ...)
    - dashboard-api 已启动(5 门全开)
    - InMemoryApprovalGateAuditStore 已注入
    """
    items = generate_path4_spike_100()
    success_count = 0
    audit_count = 0
    for item in items:
        # 调用 dashboard-api POST /api/approval-gate/actions
        response = call_dashboard_api(item)
        if response.success:
            success_count += 1
        if response.audit_id:
            audit_count += 1

    # 验证
    assert success_count == 100, f"成功数 {success_count} 应等于 100"
    assert audit_count == 100, f"audit 落档数 {audit_count} 应等于 100"
    print(f"✅ Path 4 spike 100 通过:success={success_count} audit={audit_count}")
```
