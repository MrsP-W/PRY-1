### 2.2 方案 A 工厂模式细节

```python
# 沿 docs/outlook-gmail-evaluation.md 方案 A
class SMTPProvider(Protocol):
    def send(self, message: EmailMessage) -> SendResult: ...
    def healthcheck(self) -> bool: ...

class QQProvider: ...          # 已实化,沿 D5.6.5 4 退出码
class OutlookProvider: ...    # 待实化(smtp.office365.com:587)
class GmailProvider: ...      # 待实化(smtp.gmail.com:587)

class SMTPFactory:
    @staticmethod
    def create(provider_name: str, credentials: Keychain) -> SMTPProvider:
        if provider_name == "qq": return QQProvider(credentials)
        if provider_name == "outlook": return OutlookProvider(credentials)
        if provider_name == "gmail": return GmailProvider(credentials)
        raise UnknownProviderError(provider_name)
```
